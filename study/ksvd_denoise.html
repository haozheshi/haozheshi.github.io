KSVD Method to denoise

KSVD is a sparse encoder method, To represent the signal in a sparsity way, KSVD
do two-step updates
step1: updates the coffient
step2:updataes the coffients
type coffient_update_speed_up.m
function A = coffient_update_speed_up(D, X, sigma, rc_min, max_coeff) [~,P]=size(X); [n,~]=size(D); Dict=D; W=1./sqrt(diag(Dict'*Dict)); Dict=Dict*diag(W); A_sparse_loc_row = zeros(max_coeff, P); A_sparse_loc_col = zeros(max_coeff, P); A_sparse_loc_v = zeros(max_coeff, P); parfor k=1:P x=X(:,k); residual=x; j = 0; rc_max = inf; threshold = norm(x)*sigma; DD = zeros(n, max_coeff); sparse_loc_row = zeros(max_coeff, 1); sparse_loc_col = zeros(max_coeff, 1); sparse_loc_v = zeros(max_coeff, 1); while (norm(residual)> threshold && rc_max > rc_min && j < max_coeff ) j = j+1; proj=Dict'*residual; [rc_max, pos] = max(abs(proj)); DD(:, j) = Dict(:, pos); sparse_loc_row(j) = pos; sparse_loc_col(j) = k; a=pinv(DD(:, 1 : j))*x; sparse_loc_v(1:j) = a; residual=x-DD(:, 1:j)*a; end A_sparse_loc_row(:, k) = sparse_loc_row; A_sparse_loc_col(:, k) = sparse_loc_col; A_sparse_loc_v(:, k) = sparse_loc_v; end A_sparse_loc_row = reshape(A_sparse_loc_row, 1, max_coeff * P); A_sparse_loc_col = reshape(A_sparse_loc_col, 1, max_coeff * P); A_sparse_loc_v = reshape(A_sparse_loc_v, 1, max_coeff * P); ind = A_sparse_loc_row > 0; A_sparse_loc_row = A_sparse_loc_row(ind); A_sparse_loc_col = A_sparse_loc_col(ind); A_sparse_loc_v = A_sparse_loc_v(ind); A = sparse(A_sparse_loc_row, A_sparse_loc_col, A_sparse_loc_v, size(D,2),size(X,2));
type dictionary_update.m
function [Dict, Coeff]=dictionary_update(Dict,X, Coeff) for atom=1:1:size(Dict,2) Omega=find(abs(Coeff(atom,:))>0); % this atom will be useless if isempty(Omega) continue; end CoeffM=Coeff; CoeffM(atom,:)=0; Err=(X(:,Omega)-Dict*CoeffM(:,Omega)); dd=diag(1./(ones(size(Dict, 1), length(Omega)) * (Coeff(atom,Omega).^2)'))*(Err*Coeff(atom,Omega)'); Dict(:,atom)=dd/norm(dd); Coeff(atom,Omega)=(dd'*Err)./sum(((dd*ones(1,length(Omega)))).^2,1); end end

function run(I, max_coeff, Max_iter) sigma = 0.01; rc_min = 0.01; K = 256; bb = 8; overlap = 4; I = double(I); [N,M]=size(I); NN = ceil((N-bb)/overlap) * overlap + bb; MM = ceil((M-bb)/overlap) * overlap + bb; img_new = 255 * zeros(NN,MM); img_new(1:N,1:M) = I; X = overlap_im2col(img_new, bb, overlap); DCT=zeros(bb,sqrt(K)); for k=0:1:sqrt(K)-1 V=cos([0:1:bb-1]'*k*pi/sqrt(K)); if k>0 V=V-mean(V); end DCT(:,k+1)=V/norm(V); end DCT=kron(DCT,DCT); for i = 1 : Max_iter A = coffient_update_speed_up(DCT,X,sigma,rc_min, max_coeff); [DCT, A] = dictionary_update(DCT, X, A); end Err = sum(sum((DCT* A - X).^2)); sprintf('residual error%15.2f', Err); imag_Y = overlap_col2im(DCT* A, bb, overlap, size(img_new)); imag_Y=max(min(imag_Y,255),0); img = uint8(imag_Y(1:N, 1:M)); subplot(1, 2, 1) imshow(uint8(I)) subplot(1, 2, 2) imshow(img) end
